function sesliSunum() { const usd = document.getElementById('usd').innerText.replace('₺', 'lira'); const altin = document.getElementById('q-gold').innerText.replace('₺', 'lira'); const hava = document.getElementById('weather-temp').innerText.replace('°C', 'derece'); const durum = document.getElementById('weather-info').innerText; const metin = `Sayın izleyenler, Global Insight ekonomi bültenine hoş geldiniz. Bugün Burdur'da hava sıcaklığı ${hava} ve gökyüzü ${durum}. Piyasalardaki son rakamlar ise şöyle: Dolar ${usd} seviyesinden işlem görürken, çeyrek altın fiyatı ${altin} olarak kaydedildi. Haber bültenimizi dinlediğiniz için teşekkür ederiz.`; const ses = new SpeechSynthesisUtterance(); ses.text = metin; // TÜRKÇE SESİ ZORLA BULMA SİSTEMİ const voices = window.speechSynthesis.getVoices(); const trVoice = voices.find(voice => voice.lang === 'tr-TR' || voice.lang.includes('tr')); if (trVoice) { ses.voice = trVoice; } ses.lang = 'tr-TR'; ses.rate = 1.0; window.speechSynthesis.speak(ses); } // Seslerin yüklenmesi için tarayıcıya süre tanıyan ekleme window.speechSynthesis.onvoiceschanged = () => { window.speechSynthesis.getVoices(); };
